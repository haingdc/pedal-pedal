Và CSS:

```css
@import url('https://fonts.googleapis.com/css2?family=Nunito:ital,wght@0,400;0,700;1,400;1,700&display=swap');

@layer default {
  body {
    background-color: #fff;
    font-family: Nunito;
  }
}

@layer my-components {
  .container {
    container-type: inline-size;
  }

  .row {
    display: flex;
  }

  .schedule {
    display: flex;
    flex-direction: column;
    gap: 20px;
    max-width: 800px;

    .label {
      font-size: 24px;
      line-height: 32px;
      font-weight: bold;
      text-transform: uppercase;
    }

    .text {
      font-size: 16px;
    }

    .bold {
      font-weight: bold;
    }

    .italic {
      font-style: italic;
    }

    .schedule--header {
      display: flex;

      .label {
        padding-right: 32px;
      }

      .value {
        background-color: #000;
        color: #fff;
        padding: 4px 32px;
        font-size: 16px;
        font-weight: bold;
      }

      .else {
        flex: 1;
        background-color: #fed700;
        border-radius: 0 36px 0 0;
      }
    }

    .schedule--table {
      display: grid;
      grid-template-columns: auto 1fr;

      .header {
        grid-column: 1 / 3;
        background-color: #000;
        color: #fff;
        padding: 4px 40px;
        font-size: 20px;
      }

      .cell--key,
      .cell--value {
        border-left: 1px solid #000;
        border-bottom: 1px solid #000;
      }

      .cell--key {
        background-color: #e6e7e9;
        color: #000;
        padding: 8px 20px;
      }

      .cell--value {
        padding: 8px 16px;
      }
    }
  }
}
```
